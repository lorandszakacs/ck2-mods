tribal = {

    #DoD Bloocline
	tb_dod_retinues = { #Dragon Mound
		desc = "Adds massive amounts of retinues"

		potential = {
			OR = {
				FROMFROM = { has_building = tb_dod_retinues } #Show the building only if it has been already built or if the owner has the proper bloodline to build it.
				FROM = {
					any_owned_bloodline = {
						has_bloodline_flag = dance_of_dragons_bloodline_flag
					}
				}
			}
		}

		trigger = {
			TECH_FORTIFICATIONS_CONSTRUCTION = 0
			custom_tooltip = {
				text = "Dance of the dragons Retinues"
				hidden_tooltip = {
					OR = {
						FROM = {
							any_owned_bloodline = {
								has_bloodline_flag = dance_of_dragons_bloodline_flag
							}
						}
					}
				}
			}
		}

		is_active_trigger = {
            custom_tooltip = {
                text = "This building is active only when the direct owner is of the Dance of the Dragons bloodline"
                hidden_tooltip = {
                    OR = {
                        FROM = {
                            any_owned_bloodline = {
                                has_bloodline_flag = dance_of_dragons_bloodline_flag
                            }
                        }
                    }
                }
            }
		}

		gold_cost = 50
		build_time = 200

        fort_level = 25
        tax_income = 10
        retinuesize = 5000

		add_number_to_name = no
		ai_creation_factor = 0

        convert_to_castle = ca_dod_retinues

		extra_tech_building_start = 8.0
	}

}
